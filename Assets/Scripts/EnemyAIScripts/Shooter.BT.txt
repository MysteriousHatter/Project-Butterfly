//This Behaviour Tree controls the 'shooter' unit.

tree("Root")
	//This unit attacks when possible,
	//otherwise it waits.
	fallback
		tree("Idle")
		tree("Attack")
		

tree("Idle")
	//While no enemy is spotted,
	//follow the assigned waypoints.
	while
		not IsPlayerClose
		Wait(0.2)
				

tree("Attack")
	//When the enemy is visible, attack it,
	//Otherwise forget about it.
	fallback

		//Attack the enemy if visible.
		repeat
			sequence
				IsPlayerClose
				AimAt_Player
				Wait(0.5)
				Fire
				Wait(1.0)

		//Otherwise forget about it.
		sequence
			Fail
